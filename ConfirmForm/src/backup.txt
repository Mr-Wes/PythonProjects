    def getInput(self):
        #定义返回数组：订单号，客户名称，商务，电话，颜色，丝印，型号，数量
        #主板，显示屏，二代证，指纹仪，摄像头，键盘，电缆线
        #国密，禾嘉读卡器，电池，4G，支架
        #应用程序版本，固件版本，m3版本，备注
        info = []

        info.append(self.cb_type.currentText())

        info.append(self.cb_board.currentText())
        info.append(self.cb_screen.currentText())
        info.append(self.cb_id.currentText())
        info.append(self.cb_finger.currentText())
        info.append(self.cb_camera.currentText())
        info.append('有按键板' if(self.rb_has_m3.isChecked()) else '无按键板')
        info.append(self.cb_wire.currentText())
        info.append('国密' if(self.chb_gm.isChecked()) else '无国密')
        info.append('禾嘉读卡器' if(self.chb_m1.isChecked()) else '')
        info.append('电池' if(self.chb_battery.isChecked()) else '')
        info.append('4G模块' if(self.chb_4g.isChecked()) else '')
        info.append('有支架' if(self.chb_stand.isChecked()) else '无支架')
        info.append(self.le_software.displayText())
        info.append(self.le_hardware.displayText())
        info.append(self.le_m3.displayText())
        info.append(self.te_mark.toPlainText())
        return info
